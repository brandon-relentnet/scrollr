identifier string Required Email or username

Parameter Type Required Description

WebSocket Events
WS Finance WebSocket Events
Event Type Direction Description
connection Client ‚Üí Server Initial connection handshake
filter_request Client ‚Üí Server Request filtered data with specific symbols
filtered_data Server ‚Üí Client Filtered trade data based on request
financial_update Server ‚Üí Client Real-time market updates

Code Examples

JavaScript/TypeScript 
// Authentication
const response = await fetch('https://scrollr.olvyx.com/api/accounts/auth/login', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({
    identifier: 'user@example.com',
    password: 'yourpassword'
  })
});

const { token } = await response.json();

// WebSocket Connection
const ws = new WebSocket('wss://scrollr.olvyx.com/api/finance/ws');

ws.onopen = () => {
  // Send connection message
  ws.send(JSON.stringify({
    type: 'connection',
    timestamp: Date.now()
  }));
  
  // Request filtered data
  ws.send(JSON.stringify({
    type: 'filter_request',
    filters: ['symbol_AAPL', 'symbol_GOOGL']
  }));
};

ws.onmessage = (event) => {
  const data = JSON.parse(event.data);
  console.log('Received:', data);
};
Python 
import requests
import websocket
import json

# Authentication
response = requests.post('https://scrollr.olvyx.com/api/accounts/auth/login', 
    json={
        'identifier': 'user@example.com',
        'password': 'yourpassword'
    }
)

token = response.json()['token']

# WebSocket Connection
def on_open(ws):
    # Send connection message
    ws.send(json.dumps({
        'type': 'connection',
        'timestamp': int(time.time() * 1000)
    }))
    
    # Request filtered data
    ws.send(json.dumps({
        'type': 'filter_request',
        'filters': ['symbol_AAPL', 'symbol_GOOGL']
    }))

def on_message(ws, message):
    data = json.loads(message)
    print(f'Received: {data}')

ws = websocket.WebSocketApp('wss://scrollr.olvyx.com/api/finance/ws',
    on_open=on_open,
    on_message=on_message)

ws.run_forever()
cURL 
# Login
curl -X POST https://scrollr.olvyx.com/api/accounts/auth/login \
  -H "Content-Type: application/json" \
  -d '{"identifier": "user@example.com", "password": "yourpassword"}'

# Get user profile (with token)
curl -X GET https://scrollr.olvyx.com/api/accounts/auth/me \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# Get all trades
curl -X GET https://scrollr.olvyx.com/api/finance/trades

# Get specific symbol trades
curl -X GET https://scrollr.olvyx.com/api/finance/trades/symbol/AAPL
